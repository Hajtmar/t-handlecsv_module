\unexpanded\def\lineaction{
\startxrow
    \startxcell\doiftextelse{\csvcell['A', \numline]}{\csvcell['A',
    \numline]}{empty}\stopxcell
    \startxcell
        \doiftextelse{\csvcell['A',\thenumexpr{\numline+1}]}
            {\csvcell['A',\thenumexpr{\numline+1}]}{---}
    \stopxcell
\stopxrow
}

    \filelineaction
